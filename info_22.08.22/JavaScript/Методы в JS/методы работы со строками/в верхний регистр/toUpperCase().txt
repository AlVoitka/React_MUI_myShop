toUpperCase() позволяет преобразовать строку в заглавные буквы (верхний регистр). 

Пример использования
var str = "crocodile", // инициализируем строковые переменные
    str2 =  "1a2b3c"; 

var newString = str.toUpperCase(); // переменная содержит значение "CROCODILE"
var newString2 = str2.toUpperCase(); // переменная содержит значение "1A2B3C"
